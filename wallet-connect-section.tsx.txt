'use client';

import { BaseWallet } from './base-wallet';
import { BaseSignIn } from '../base-sign-in';
import { useMiniKit } from '@/hooks/use-minikit';
import { useAccount } from 'wagmi';
import { isUserWhitelisted } from '@/lib/whitelist';
import { Badge } from '../ui/badge';
import { Sparkles, Wallet, Users } from 'lucide-react';

/**
 * Unified Wallet Connection Section
 * Shows appropriate auth method based on context
 */
export function WalletConnectSection(): JSX.Element {
  const { isInMiniApp, user: miniKitUser, loading } = useMiniKit();
  const { address, isConnected } = useAccount();

  // Check if user is whitelisted
  const whitelisted = isUserWhitelisted({
    fid: miniKitUser?.fid,
    evmAddress: address,
  });

  if (loading) {
    return (
      <div className="flex items-center gap-2 px-4 py-2 rounded-lg bg-gray-100">
        <div className="animate-pulse flex items-center gap-2">
          <div className="w-8 h-8 bg-gray-300 rounded-full" />
          <div className="w-24 h-4 bg-gray-300 rounded" />
        </div>
      </div>
    );
  }

  return (
    <div className="flex items-center gap-3">
      {/* Mini App Context - Show FID */}
      {isInMiniApp && miniKitUser && (
        <div className="flex items-center gap-2 px-3 py-1.5 rounded-lg bg-purple-100 border border-purple-200">
          <Users className="w-4 h-4 text-purple-600" />
          <span className="text-sm font-medium text-purple-900">
            {miniKitUser.displayName || miniKitUser.username || `FID ${miniKitUser.fid}`}
          </span>
        </div>
      )}

      {/* Base Builder Context - Show Base Sign In */}
      {!isConnected && typeof window !== 'undefined' && (window as any).baseAccount && (
        <BaseSignIn align="center" variant="solid" colorScheme="light" />
      )}

      {/* Wallet Connection - Show Coinbase Wallet */}
      {!isInMiniApp && <BaseWallet />}

      {/* Whitelisted Badge */}
      {whitelisted && (
        <Badge variant="secondary" className="flex items-center gap-1 bg-green-100 text-green-700 border-green-200">
          <Sparkles className="w-3 h-3" />
          FREE ACCESS
        </Badge>
      )}

      {/* Connected Wallet Badge */}
      {isConnected && !whitelisted && (
        <Badge variant="secondary" className="flex items-center gap-1 bg-blue-100 text-blue-700 border-blue-200">
          <Wallet className="w-3 h-3" />
          Connected
        </Badge>
      )}
    </div>
  );
}
