'use client';

import { useState } from 'react';
import { Dialog, DialogContent, DialogDescription, DialogHeader, DialogTitle } from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Textarea } from '@/components/ui/textarea';
import { Label } from '@/components/ui/label';
import { Input } from '@/components/ui/input';
import { Share2, Check, ExternalLink } from 'lucide-react';

interface LensPostModalProps {
  open: boolean;
  onClose: () => void;
  userId: string;
  paymentId: string;
}

interface PostResult {
  postId: string;
  transactionHash: string;
  lensUrl: string;
  status: string;
}

export function LensPostModal({ open, onClose, userId, paymentId }: LensPostModalProps): JSX.Element {
  const [content, setContent] = useState<string>('');
  const [mediaUrl, setMediaUrl] = useState<string>('');
  const [loading, setLoading] = useState<boolean>(false);
  const [result, setResult] = useState<PostResult | null>(null);

  const handlePost = async (): Promise<void> => {
    if (!content.trim()) {
      alert('Please enter post content');
      return;
    }

    setLoading(true);

    try {
      const response = await fetch('/api/lens/post', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          userId,
          paymentId,
          content: content.trim(),
          mediaUrl: mediaUrl || undefined,
        }),
      });

      if (response.ok) {
        const data = await response.json();
        setResult(data);
      } else {
        const error = await response.json();
        alert(error.error || 'Lens post failed');
      }
    } catch (error) {
      console.error('Lens post error:', error);
      alert('Post failed');
    } finally {
      setLoading(false);
    }
  };

  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent className="bg-gray-900 text-white border-purple-500/30 max-w-2xl">
        <DialogHeader>
          <DialogTitle>Lens Post - Decentralized Social</DialogTitle>
          <DialogDescription className="text-gray-400">
            Publish onchain content to Lens Protocol
          </DialogDescription>
        </DialogHeader>

        {!result ? (
          <div className="space-y-4">
            <div className="bg-blue-900/20 border border-blue-500/30 rounded-lg p-4">
              <p className="text-blue-300 text-sm">
                <strong>Lens Protocol:</strong> Your content is stored onchain and owned by you forever
              </p>
            </div>

            <div>
              <Label htmlFor="content">Post Content *</Label>
              <Textarea
                id="content"
                value={content}
                onChange={(e) => setContent(e.target.value)}
                placeholder="What's happening in Web3?"
                rows={6}
                maxLength={500}
                className="bg-gray-800 border-gray-700"
              />
              <p className="text-xs text-gray-500 mt-1 text-right">
                {content.length}/500 characters
              </p>
            </div>

            <div>
              <Label htmlFor="mediaUrl">Media URL (Optional)</Label>
              <Input
                id="mediaUrl"
                value={mediaUrl}
                onChange={(e) => setMediaUrl(e.target.value)}
                placeholder="https://... (image, video, or IPFS link)"
                className="bg-gray-800 border-gray-700"
              />
            </div>

            <Button
              onClick={handlePost}
              disabled={loading || !content.trim()}
              className="w-full bg-purple-600 hover:bg-purple-700"
            >
              {loading ? 'Publishing to Lens...' : 'Publish Post'}
            </Button>
          </div>
        ) : (
          <div className="space-y-4">
            <div className="bg-green-900/20 border border-green-500/30 rounded-lg p-4">
              <div className="flex items-center gap-2 text-green-400 mb-2">
                <Check className="w-5 h-5" />
                <span className="font-semibold">Posted to Lens Successfully!</span>
              </div>
              <p className="text-gray-300 text-sm">
                Your content is now onchain and visible across all Lens apps
              </p>
            </div>

            <div className="space-y-3">
              <div className="bg-gray-800/50 rounded-lg p-4">
                <Label className="text-sm text-gray-400">Post ID</Label>
                <p className="text-white text-sm font-mono break-all mt-1">{result.postId}</p>
              </div>

              <div className="bg-gray-800/50 rounded-lg p-4">
                <Label className="text-sm text-gray-400">Transaction Hash</Label>
                <p className="text-white text-sm font-mono break-all mt-1">{result.transactionHash}</p>
              </div>

              <div className="bg-gray-800/50 rounded-lg p-4">
                <Label className="text-sm text-gray-400">Status</Label>
                <p className="text-white text-sm mt-1 capitalize">{result.status}</p>
              </div>

              <a
                href={result.lensUrl}
                target="_blank"
                rel="noopener noreferrer"
                className="block"
              >
                <Button className="w-full bg-green-600 hover:bg-green-700">
                  View on Lens
                  <ExternalLink className="w-4 h-4 ml-2" />
                </Button>
              </a>
            </div>

            <div className="bg-blue-900/20 border border-blue-500/30 rounded-lg p-4">
              <p className="text-blue-300 text-sm">
                <strong>Your post is now live on:</strong>
              </p>
              <ul className="text-blue-300 text-sm mt-2 space-y-1 list-disc list-inside">
                <li>Lens.xyz</li>
                <li>Hey.xyz (Lenster)</li>
                <li>Orb.ac</li>
                <li>All other Lens-compatible apps</li>
              </ul>
            </div>

            <Button
              onClick={() => {
                setResult(null);
                setContent('');
                setMediaUrl('');
              }}
              variant="outline"
              className="w-full border-gray-700 hover:bg-gray-800"
            >
              Create Another Post
            </Button>
          </div>
        )}
      </DialogContent>
    </Dialog>
  );
}
