'use client';

import { useEffect, useState } from 'react';
import { useFarcasterAuth } from '@/hooks/use-farcaster-auth';
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Textarea } from '@/components/ui/textarea';
import { Badge } from '@/components/ui/badge';
import { Trophy, Clock, DollarSign, ArrowLeft, ThumbsUp } from 'lucide-react';
import Link from 'next/link';
import type { Contest } from '@/types/app';

interface ContestEntry {
  id: string;
  user_id: string;
  content: string;
  voteCount: number;
  user: { fid: number; username: string | null } | null;
}

export default function ContestsPage(): JSX.Element {
  const { userId, loading } = useFarcasterAuth();
  const [contests, setContests] = useState<Contest[]>([]);
  const [selectedContest, setSelectedContest] = useState<Contest | null>(null);
  const [entries, setEntries] = useState<ContestEntry[]>([]);
  const [newEntry, setNewEntry] = useState<string>('');
  const [submitting, setSubmitting] = useState<boolean>(false);
  const [voting, setVoting] = useState<boolean>(false);

  useEffect(() => {
    loadContests();
  }, []);

  const loadContests = async (): Promise<void> => {
    try {
      const res = await fetch('/api/contests/list?status=active');
      const data = await res.json();
      setContests(data.contests || []);
    } catch (error) {
      console.error('Failed to load contests:', error);
    }
  };

  const loadEntries = async (contestId: string): Promise<void> => {
    try {
      const res = await fetch(`/api/contests/entries?contestId=${contestId}`);
      const data = await res.json();
      setEntries(data.entries || []);
    } catch (error) {
      console.error('Failed to load entries:', error);
    }
  };

  const handleSelectContest = (contest: Contest): void => {
    setSelectedContest(contest);
    loadEntries(contest.id);
  };

  const handleSubmitEntry = async (): Promise<void> => {
    if (!selectedContest || !userId || !newEntry.trim()) {
      alert('Please fill in your entry');
      return;
    }

    setSubmitting(true);
    try {
      const res = await fetch('/api/contests/enter', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contestId: selectedContest.id,
          userId,
          contentType: 'text',
          content: newEntry,
          entryFeeUsd: selectedContest.entry_fee_usd,
          chain: 'Base',
          txHash: `mock_tx_${Date.now()}`, // Replace with real payment
        }),
      });

      if (res.ok) {
        alert('Entry submitted successfully!');
        setNewEntry('');
        loadEntries(selectedContest.id);
      } else {
        const error = await res.json();
        alert(error.error || 'Failed to submit entry');
      }
    } catch (error) {
      console.error('Submit entry error:', error);
      alert('Failed to submit entry');
    } finally {
      setSubmitting(false);
    }
  };

  const handleVote = async (entryId: string): Promise<void> => {
    if (!selectedContest || !userId) return;

    setVoting(true);
    try {
      const res = await fetch('/api/contests/vote', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contestId: selectedContest.id,
          userId,
          entryId,
        }),
      });

      if (res.ok) {
        alert('Vote submitted!');
        loadEntries(selectedContest.id);
      } else {
        const error = await res.json();
        alert(error.error || 'Failed to vote');
      }
    } catch (error) {
      console.error('Vote error:', error);
      alert('Failed to vote');
    } finally {
      setVoting(false);
    }
  };

  if (loading) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-gray-900 via-purple-900 to-black flex items-center justify-center">
        <p className="text-white">Loading...</p>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-gray-900 via-purple-900 to-black pt-16 pb-12 px-4">
      <div className="max-w-6xl mx-auto">
        <div className="mb-6">
          <Link href="/">
            <Button variant="ghost" className="text-white hover:bg-purple-800/50">
              <ArrowLeft className="w-4 h-4 mr-2" />
              Back to Home
            </Button>
          </Link>
        </div>

        <header className="text-center mb-12">
          <h1 className="text-5xl font-bold text-transparent bg-clip-text bg-gradient-to-r from-purple-400 to-pink-400 mb-4">
            Contests
          </h1>
          <p className="text-gray-300 text-lg">
            Compete for prizes with your memecoins, memes, and NFT ideas
          </p>
        </header>

        {!selectedContest ? (
          <div className="space-y-4">
            {contests.length === 0 ? (
              <Card className="bg-gray-900/50 border-purple-500/30">
                <CardContent className="py-12 text-center">
                  <Trophy className="w-16 h-16 text-gray-600 mx-auto mb-4" />
                  <p className="text-gray-400 text-lg">No active contests yet. Check back soon!</p>
                </CardContent>
              </Card>
            ) : (
              contests.map((contest) => (
                <Card
                  key={contest.id}
                  className="bg-gray-900/50 border-purple-500/30 hover:border-purple-400 transition-colors cursor-pointer"
                  onClick={() => handleSelectContest(contest)}
                >
                  <CardHeader>
                    <div className="flex items-start justify-between">
                      <div>
                        <CardTitle className="text-white text-xl">{contest.title}</CardTitle>
                        <CardDescription className="text-gray-400 mt-2">
                          {contest.description}
                        </CardDescription>
                      </div>
                      <Badge className="bg-purple-600">{contest.type}</Badge>
                    </div>
                  </CardHeader>
                  <CardContent>
                    <div className="flex items-center gap-6 text-sm">
                      <div className="flex items-center gap-2 text-green-400">
                        <DollarSign className="w-4 h-4" />
                        <span>${contest.total_pot_usd} Prize Pool</span>
                      </div>
                      <div className="flex items-center gap-2 text-yellow-400">
                        <DollarSign className="w-4 h-4" />
                        <span>${contest.entry_fee_usd} Entry</span>
                      </div>
                      <div className="flex items-center gap-2 text-blue-400">
                        <Clock className="w-4 h-4" />
                        <span>Ends {new Date(contest.ends_at).toLocaleDateString()}</span>
                      </div>
                    </div>
                  </CardContent>
                </Card>
              ))
            )}
          </div>
        ) : (
          <div className="space-y-6">
            <Button
              onClick={() => setSelectedContest(null)}
              variant="ghost"
              className="text-white hover:bg-purple-800/50"
            >
              <ArrowLeft className="w-4 h-4 mr-2" />
              Back to Contests
            </Button>

            <Card className="bg-gray-900/50 border-purple-500/30">
              <CardHeader>
                <CardTitle className="text-white text-2xl">{selectedContest.title}</CardTitle>
                <CardDescription className="text-gray-400">{selectedContest.description}</CardDescription>
              </CardHeader>
              <CardContent className="space-y-4">
                <div className="grid grid-cols-3 gap-4 text-center">
                  <div>
                    <p className="text-2xl font-bold text-green-400">${selectedContest.total_pot_usd}</p>
                    <p className="text-gray-400 text-sm">Prize Pool</p>
                  </div>
                  <div>
                    <p className="text-2xl font-bold text-yellow-400">${selectedContest.entry_fee_usd}</p>
                    <p className="text-gray-400 text-sm">Entry Fee</p>
                  </div>
                  <div>
                    <p className="text-2xl font-bold text-blue-400">{entries.length}</p>
                    <p className="text-gray-400 text-sm">Entries</p>
                  </div>
                </div>

                <div className="border-t border-gray-700 pt-4">
                  <h3 className="text-white font-semibold mb-3">Submit Your Entry</h3>
                  <Textarea
                    value={newEntry}
                    onChange={(e) => setNewEntry(e.target.value)}
                    placeholder="Enter your contest submission..."
                    rows={4}
                    className="bg-gray-800 border-gray-700 text-white mb-3"
                  />
                  <Button
                    onClick={handleSubmitEntry}
                    disabled={submitting || !userId}
                    className="w-full bg-purple-600 hover:bg-purple-700"
                  >
                    {submitting ? 'Submitting...' : `Submit Entry ($${selectedContest.entry_fee_usd})`}
                  </Button>
                </div>
              </CardContent>
            </Card>

            <Card className="bg-gray-900/50 border-purple-500/30">
              <CardHeader>
                <CardTitle className="text-white">Contest Entries</CardTitle>
                <CardDescription className="text-gray-400">Vote for your favorite entries</CardDescription>
              </CardHeader>
              <CardContent>
                {entries.length === 0 ? (
                  <p className="text-gray-400 text-center py-8">No entries yet. Be the first!</p>
                ) : (
                  <div className="space-y-3">
                    {entries.map((entry) => (
                      <div
                        key={entry.id}
                        className="p-4 bg-gray-800/50 rounded-lg border border-gray-700"
                      >
                        <div className="flex items-start justify-between mb-2">
                          <div>
                            <p className="text-white font-semibold">
                              {entry.user?.username || `FID ${entry.user?.fid}`}
                            </p>
                          </div>
                          <div className="flex items-center gap-2">
                            <Badge className="bg-purple-600">
                              {entry.voteCount} {entry.voteCount === 1 ? 'vote' : 'votes'}
                            </Badge>
                            <Button
                              size="sm"
                              onClick={() => handleVote(entry.id)}
                              disabled={voting || !userId || entry.user_id === userId}
                              className="bg-pink-600 hover:bg-pink-700"
                            >
                              <ThumbsUp className="w-4 h-4" />
                            </Button>
                          </div>
                        </div>
                        <p className="text-gray-300 whitespace-pre-wrap">{entry.content}</p>
                      </div>
                    ))}
                  </div>
                )}
              </CardContent>
            </Card>
          </div>
        )}
      </div>
    </div>
  );
}
