'use client';

import { useEffect, useState } from 'react';
import { useFarcasterAuth } from '@/hooks/use-farcaster-auth';
import { Card, CardContent, CardDescription, CardHeader, CardTitle } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Badge } from '@/components/ui/badge';
import { Copy, DollarSign, Users, TrendingUp, ArrowLeft } from 'lucide-react';
import Link from 'next/link';

interface Earning {
  id: string;
  amount_usd: number;
  paid_out: boolean;
  created_at: string;
  referred: { fid: number; username: string | null } | null;
  payment: { usd_equiv: number; feature_type: string; created_at: string } | null;
}

export default function ReferralsPage(): JSX.Element {
  const { userId, loading } = useFarcasterAuth();
  const [referralCode, setReferralCode] = useState<string>('');
  const [earnings, setEarnings] = useState<Earning[]>([]);
  const [stats, setStats] = useState({ totalEarned: 0, totalPaidOut: 0, pending: 0 });
  const [generatingCode, setGeneratingCode] = useState<boolean>(false);

  useEffect(() => {
    if (userId) {
      loadReferralData();
    }
  }, [userId]);

  const loadReferralData = async (): Promise<void> => {
    try {
      // Generate or get referral code
      setGeneratingCode(true);
      const codeRes = await fetch('/api/referrals/generate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ userId }),
      });
      const codeData = await codeRes.json();
      setReferralCode(codeData.code);

      // Get earnings
      const earningsRes = await fetch(`/api/referrals/earnings?userId=${userId}`);
      const earningsData = await earningsRes.json();
      setEarnings(earningsData.earnings || []);
      setStats({
        totalEarned: earningsData.totalEarned || 0,
        totalPaidOut: earningsData.totalPaidOut || 0,
        pending: earningsData.pending || 0,
      });
    } catch (error) {
      console.error('Failed to load referral data:', error);
    } finally {
      setGeneratingCode(false);
    }
  };

  const copyReferralLink = async (): Promise<void> => {
    const link = `${window.location.origin}/ref/${referralCode}`;
    await navigator.clipboard.writeText(link);
    alert('Referral link copied!');
  };

  const copyCode = async (): Promise<void> => {
    await navigator.clipboard.writeText(referralCode);
    alert('Referral code copied!');
  };

  if (loading) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-gray-900 via-purple-900 to-black flex items-center justify-center">
        <p className="text-white">Loading...</p>
      </div>
    );
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-gray-900 via-purple-900 to-black pt-16 pb-12 px-4">
      <div className="max-w-6xl mx-auto">
        <div className="mb-6">
          <Link href="/">
            <Button variant="ghost" className="text-white hover:bg-purple-800/50">
              <ArrowLeft className="w-4 h-4 mr-2" />
              Back to Home
            </Button>
          </Link>
        </div>

        <header className="text-center mb-12">
          <h1 className="text-5xl font-bold text-transparent bg-clip-text bg-gradient-to-r from-purple-400 to-pink-400 mb-4">
            Referral Dashboard
          </h1>
          <p className="text-gray-300 text-lg">Earn 5-20% commission on every referral</p>
        </header>

        <div className="grid grid-cols-1 md:grid-cols-3 gap-6 mb-8">
          <Card className="bg-gray-900/50 border-purple-500/30">
            <CardHeader>
              <CardTitle className="flex items-center gap-2 text-white">
                <DollarSign className="w-5 h-5 text-green-400" />
                Total Earned
              </CardTitle>
            </CardHeader>
            <CardContent>
              <p className="text-3xl font-bold text-green-400">${stats.totalEarned.toFixed(2)}</p>
            </CardContent>
          </Card>

          <Card className="bg-gray-900/50 border-purple-500/30">
            <CardHeader>
              <CardTitle className="flex items-center gap-2 text-white">
                <TrendingUp className="w-5 h-5 text-yellow-400" />
                Pending
              </CardTitle>
            </CardHeader>
            <CardContent>
              <p className="text-3xl font-bold text-yellow-400">${stats.pending.toFixed(2)}</p>
            </CardContent>
          </Card>

          <Card className="bg-gray-900/50 border-purple-500/30">
            <CardHeader>
              <CardTitle className="flex items-center gap-2 text-white">
                <Users className="w-5 h-5 text-blue-400" />
                Referrals
              </CardTitle>
            </CardHeader>
            <CardContent>
              <p className="text-3xl font-bold text-blue-400">{earnings.length}</p>
            </CardContent>
          </Card>
        </div>

        <Card className="bg-gray-900/50 border-purple-500/30 mb-8">
          <CardHeader>
            <CardTitle className="text-white">Your Referral Code</CardTitle>
            <CardDescription className="text-gray-400">
              Share your code and earn commission on every purchase
            </CardDescription>
          </CardHeader>
          <CardContent className="space-y-4">
            <div className="flex gap-2">
              <Input
                value={referralCode || 'Generating...'}
                readOnly
                className="bg-gray-800 border-gray-700 text-white font-mono text-lg"
              />
              <Button
                onClick={copyCode}
                disabled={!referralCode || generatingCode}
                className="bg-purple-600 hover:bg-purple-700"
              >
                <Copy className="w-4 h-4" />
              </Button>
            </div>
            <Button
              onClick={copyReferralLink}
              disabled={!referralCode || generatingCode}
              className="w-full bg-gradient-to-r from-purple-600 to-pink-600 hover:from-purple-700 hover:to-pink-700"
            >
              Copy Referral Link
            </Button>
          </CardContent>
        </Card>

        <Card className="bg-gray-900/50 border-purple-500/30">
          <CardHeader>
            <CardTitle className="text-white">Earnings History</CardTitle>
            <CardDescription className="text-gray-400">
              Track your referral commissions
            </CardDescription>
          </CardHeader>
          <CardContent>
            {earnings.length === 0 ? (
              <p className="text-gray-400 text-center py-8">No earnings yet. Start sharing your referral link!</p>
            ) : (
              <div className="space-y-3">
                {earnings.map((earning) => (
                  <div
                    key={earning.id}
                    className="flex items-center justify-between p-4 bg-gray-800/50 rounded-lg border border-gray-700"
                  >
                    <div>
                      <p className="text-white font-semibold">
                        {earning.referred?.username || `FID ${earning.referred?.fid}`}
                      </p>
                      <p className="text-gray-400 text-sm">
                        {earning.payment?.feature_type} â€¢ {new Date(earning.created_at).toLocaleDateString()}
                      </p>
                    </div>
                    <div className="text-right">
                      <p className="text-green-400 font-bold">${earning.amount_usd.toFixed(2)}</p>
                      <Badge variant={earning.paid_out ? 'default' : 'secondary'}>
                        {earning.paid_out ? 'Paid' : 'Pending'}
                      </Badge>
                    </div>
                  </div>
                ))}
              </div>
            )}
          </CardContent>
        </Card>
      </div>
    </div>
  );
}
