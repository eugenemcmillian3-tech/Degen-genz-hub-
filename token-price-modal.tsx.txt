'use client';

import { useState } from 'react';
import { Dialog, DialogContent, DialogDescription, DialogHeader, DialogTitle } from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select';
import { DollarSign, TrendingUp, RefreshCw } from 'lucide-react';

interface TokenPriceModalProps {
  open: boolean;
  onClose: () => void;
  userId: string;
  paymentId: string;
}

interface PriceData {
  symbol: string;
  price: string;
  change24h: number;
  high24h: string;
  low24h: string;
  volume24h: string;
  marketCap: string;
  chain: string;
  lastUpdate: string;
}

export function TokenPriceModal({ open, onClose, userId, paymentId }: TokenPriceModalProps): JSX.Element {
  const [tokenAddress, setTokenAddress] = useState<string>('');
  const [chain, setChain] = useState<string>('ethereum');
  const [loading, setLoading] = useState<boolean>(false);
  const [result, setResult] = useState<PriceData | null>(null);

  const handleGetPrice = async (): Promise<void> => {
    if (!tokenAddress) {
      alert('Please enter a token address');
      return;
    }

    setLoading(true);

    try {
      const response = await fetch('/api/token/price', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          userId,
          paymentId,
          tokenAddress,
          chain,
        }),
      });

      if (response.ok) {
        const data = await response.json();
        setResult(data);
      } else {
        const error = await response.json();
        alert(error.error || 'Price lookup failed');
      }
    } catch (error) {
      console.error('Token price error:', error);
      alert('Price lookup failed');
    } finally {
      setLoading(false);
    }
  };

  return (
    <Dialog open={open} onOpenChange={onClose}>
      <DialogContent className="bg-gray-900 text-white border-purple-500/30 max-w-2xl">
        <DialogHeader>
          <DialogTitle>Token Price Lookup - 0x API</DialogTitle>
          <DialogDescription className="text-gray-400">
            Get real-time token prices and market data
          </DialogDescription>
        </DialogHeader>

        {!result ? (
          <div className="space-y-4">
            <div>
              <Label htmlFor="chain">Blockchain</Label>
              <Select value={chain} onValueChange={setChain}>
                <SelectTrigger className="bg-gray-800 border-gray-700">
                  <SelectValue />
                </SelectTrigger>
                <SelectContent className="bg-gray-800 border-gray-700">
                  <SelectItem value="ethereum">Ethereum</SelectItem>
                  <SelectItem value="base">Base</SelectItem>
                  <SelectItem value="polygon">Polygon</SelectItem>
                  <SelectItem value="arbitrum">Arbitrum</SelectItem>
                  <SelectItem value="optimism">Optimism</SelectItem>
                </SelectContent>
              </Select>
            </div>

            <div>
              <Label htmlFor="tokenAddress">Token Address</Label>
              <Input
                id="tokenAddress"
                value={tokenAddress}
                onChange={(e) => setTokenAddress(e.target.value)}
                placeholder="0x..."
                className="bg-gray-800 border-gray-700 font-mono"
              />
              <p className="text-xs text-gray-500 mt-1">
                Enter the token contract address
              </p>
            </div>

            <div className="bg-blue-900/20 border border-blue-500/30 rounded-lg p-3">
              <p className="text-blue-300 text-sm">
                <strong>Popular tokens:</strong> USDC, WETH, DAI, or any ERC-20 token address
              </p>
            </div>

            <Button
              onClick={handleGetPrice}
              disabled={loading || !tokenAddress}
              className="w-full bg-purple-600 hover:bg-purple-700"
            >
              {loading ? 'Fetching Price...' : 'Get Price'}
            </Button>
          </div>
        ) : (
          <div className="space-y-4">
            <div className="bg-gradient-to-r from-purple-900/50 to-pink-900/50 border border-purple-500/30 rounded-lg p-6">
              <div className="flex items-center justify-between mb-4">
                <div>
                  <h3 className="text-2xl font-bold text-white">{result.symbol}</h3>
                  <p className="text-gray-400 text-sm capitalize">{result.chain}</p>
                </div>
                <DollarSign className="w-8 h-8 text-purple-400" />
              </div>
              <div className="text-center">
                <p className="text-4xl font-bold text-white mb-2">${result.price}</p>
                <div className={`inline-flex items-center gap-1 ${result.change24h >= 0 ? 'text-green-400' : 'text-red-400'}`}>
                  <TrendingUp className="w-4 h-4" />
                  <span className="font-semibold">
                    {result.change24h >= 0 ? '+' : ''}{result.change24h.toFixed(2)}% (24h)
                  </span>
                </div>
              </div>
            </div>

            <div className="grid grid-cols-2 gap-4">
              <div className="bg-gray-800/50 rounded-lg p-4">
                <p className="text-gray-400 text-sm">24h High</p>
                <p className="text-white font-semibold mt-1">${result.high24h}</p>
              </div>
              <div className="bg-gray-800/50 rounded-lg p-4">
                <p className="text-gray-400 text-sm">24h Low</p>
                <p className="text-white font-semibold mt-1">${result.low24h}</p>
              </div>
              <div className="bg-gray-800/50 rounded-lg p-4">
                <p className="text-gray-400 text-sm">24h Volume</p>
                <p className="text-white font-semibold mt-1">${result.volume24h}</p>
              </div>
              <div className="bg-gray-800/50 rounded-lg p-4">
                <p className="text-gray-400 text-sm">Market Cap</p>
                <p className="text-white font-semibold mt-1">${result.marketCap}</p>
              </div>
            </div>

            <p className="text-gray-500 text-xs text-center">
              Last updated: {new Date(result.lastUpdate).toLocaleString()}
            </p>

            <div className="flex gap-2">
              <Button
                onClick={() => handleGetPrice()}
                variant="outline"
                className="flex-1 border-gray-700 hover:bg-gray-800"
              >
                <RefreshCw className="w-4 h-4 mr-2" />
                Refresh
              </Button>
              <Button
                onClick={() => {
                  setResult(null);
                  setTokenAddress('');
                }}
                variant="outline"
                className="flex-1 border-gray-700 hover:bg-gray-800"
              >
                New Lookup
              </Button>
            </div>
          </div>
        )}
      </DialogContent>
    </Dialog>
  );
}
